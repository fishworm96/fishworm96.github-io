```javascript
var arr = [5, 4, 3, 2, 1]
for (var i = 0; i <= arr.length -1; i++) {//外层循环管趟数
	for (var j = 0; j <= arr.length -i -1; j++)//里面的循环管每趟的交换次数
  if(arr[j]) > arr[j + 1] {
		var temp = arr[j];
		arr[j] = arr[j + 1];
		arr[j + 1] = temp;
	}
}
```
